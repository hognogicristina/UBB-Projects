// Compute the sum of n numbers, computer the max/min of n numbers

n: int
sum: int
i: int

sys.read(n)
sum = 0
i = 1

// Compute the sum of n numbers
while i <= n do
begin:
	sum = sum + i
	i = i + 1
end;

sys.write("Sum: ")
sys.write(sum)

#nr: arr(int) [10]
max: int
min: int

sys.read(#nr)
max = -999999
min = 999999

// Compute the max/min of n numbers
foreach num in #nr
begin:
	if num > max
	begin:
		max = num
	end;

	if num < min
	begin:
		min = num
	end;
end;

sys.write("Max: ")
sys.write(max)
sys.write("Min: ")
sys.write(min)
